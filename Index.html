/** * โค้ดส่วนหลังบ้าน (Backend) ทำหน้าที่ส่งหน้าเว็บ HTML * และดึงข้อมูลจาก Google Sheets ส่งไปให้หน้าเว็บ */ // ฟังก์ชันหลักสำหรับเปิดหน้าเว็บ function doGet(e) { return HtmlService.createHtmlOutputFromFile('Index') .setTitle('สถานการณ์โรคติดต่อจังหวัดอุดรธานี') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL) .addMetaTag('viewport', 'width=device-width, initial-scale=1'); } // ฟังก์ชันสำหรับให้หน้าเว็บเรียกใช้ เพื่อดึงข้อมูลทั้งหมด function getDashboardData() { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const responseData = { settings: getSettingsData(ss.getSheetByName("Settings")), importantDiseases: getDiseaseData(ss.getSheetByName("Important")), specialDiseases: getDiseaseData(ss.getSheetByName("Special")), eradicationDiseases: getDiseaseData(ss.getSheetByName("Eradication")), notes: getNotesData(ss.getSheetByName("Notes")) }; return { status: 'success', data: responseData }; } catch (error) { return { status: 'error', message: error.toString() }; } } // --------------------------------------------------------- // ฟังก์ชันตัวช่วยดึงข้อมูล (Helper Functions) // --------------------------------------------------------- function getSettingsData(sheet) { if (!sheet) return {}; const data = sheet.getDataRange().getValues(); const settings = {}; for (let i = 1; i < data.length; i++) { let key = data[i][0]; let value = data[i][1]; if (key) settings[key] = value; } return settings; } function getDiseaseData(sheet) { if (!sheet) return []; const data = sheet.getDataRange().getValues(); const diseases = []; for (let i = 1; i < data.length; i++) { let row = data[i]; if (!row[1]) continue; let trendDataString = String(row[6]).trim(); let trendDataArray = trendDataString ? trendDataString.split(',').map(Number) : []; let districtCases = []; for (let d = 7; d < 27; d++) { districtCases.push(Number(row[d]) || 0); } diseases.push({ id: Number(row[0]) || i, name: String(row[1]), ytdCases: Number(row[2]) || 0, ytdDeaths: Number(row[3]) || 0, weeklyCases: Number(row[4]) || 0, trend: String(row[5]).toLowerCase(), trendData: trendDataArray, cases: districtCases }); } return diseases; } function getNotesData(sheet) { if (!sheet) return []; const data = sheet.getDataRange().getValues(); const notes = []; for (let i = 1; i < data.length; i++) { let type = data[i][0]; let message = data[i][1]; if (message) { notes.push({ type: String(type).toLowerCase(), message: String(message) }); } } return notes; }
